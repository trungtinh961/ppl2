
            int main() {
                int arms ; 
                int check, rem, sum ;

                check = arms;

                do {
                    rem = check % 10;
                    sum = sum + (rem * rem * rem);
                    check = check / 10;
                } while(check != 0);

                if(sum == arms) 
                    printf("So %d la mot so armstrong.", arms);
                else 
                    printf("So %d khong phai la so armstrong.", arms);

                return 0;
            }
