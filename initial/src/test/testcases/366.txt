        // ham de trao doi gia tri
        void swap(int num1, int num2){
            temp = intArray[num1];
            intArray[num1] = intArray[num2];
            intArray[num2] = temp;
        }

        // ham de chia mang thanh 2 phan, su dung phan tu chot (pivot)
        int partition(int left, int right, int pivot){
            leftPointer = left -1;
            rightPointer = right;

            do {
                
                do {
                    //khong lam gi
                } while(intArray[1+leftPointer] < pivot);
                    
                do {
                    //khong lam gi
                } while(rightPointer > 0 && intArray[--rightPointer] > pivot);

                if(leftPointer >= rightPointer){
                    break;
                }else{
                    printf(" Phan tu duoc trao doi :%d,%d\n", 
                    intArray[leftPointer],intArray[rightPointer]);
                    swap(leftPointer,rightPointer);
                }
                    
            } while(true);
                
            printf(" Phan tu chot duoc trao doi :%d,%d\n", intArray[leftPointer],intArray[right]);
            swap(leftPointer,right);
            printf("Hien thi mang sau moi lan trao doi: "); 
            display();
            return leftPointer;
        }

        // ham sap xep
        void quickSort(int left, int right){        
            if(right-left <= 0){
                return;   
            }else {
                pivot = intArray[right];
                partitionPoint = partition(left, right, pivot);
                quickSort(left,partitionPoint-1);
                quickSort(partitionPoint+1,right);
            }        
        }
